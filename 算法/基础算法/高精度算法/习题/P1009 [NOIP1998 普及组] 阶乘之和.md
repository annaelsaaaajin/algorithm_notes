# Code
```cpp
#include <iostream>
#include <vector>

using namespace std;

vector<int> add(vector<int> A, vector<int> B)
{
    vector<int> C;

    int t = 0;
    for (int i = 0; i < A.size() || i < B.size(); i ++)
    {
        if (i < A.size()) t += A[i];
        if (i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }

    if (t) C.push_back(1);
    return C;
}

vector<int> mul_l(vector<int> &A, int b)
{
    if (b == 0 || (A.size() == 1 && A[0] == 0)) return {0};
    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size(); i ++)
    {
        t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    while (t) C.push_back(t % 10), t /= 10;
    return C;
}

vector<int> fact(int x)
{
    vector<int> res = {1};
    for (int i = 1; i <= x; i ++)
        res = mul_l(res, i);
    return res;
}

int main()
{
    vector<int> sum;
    int x;
    cin >> x;
    sum.push_back(0);
    for (int i = 1; i <= x; i++)
    {
        auto t = fact(i);
        sum = add(sum, t);
    }

    for (int i = sum.size() - 1; i >= 0; i --) printf("%d", sum[i]);

    return 0;
}
```
