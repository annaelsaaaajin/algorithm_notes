# 解题思路
- 循环链表：取模，将实数进行分类，将全体实数映射到给定区间内
```cpp
//R —Classify—> [0, R]
int num_mapping_from_zero(int r, int n)
{
    int len = r + 1;
    if (n >= 0) return n % len; //正数直接 mod
    else
    {
        return (n % len + len) % len;
        /*
        //负数取模，向0取整所以其结果要补一个len，使余数r为正
        if (n % len == 0) return 0; //特判如果能整除，余数是 0 而不是 len
        return n % len + len; //补len
        */
    }
}

//R —Classify—> [l, R]
int num_mapping(int l, int r, int n)
{
    //左端点偏移
    int t = n - l;
    return num_mapping_from_zero(r - l, t) + l;
}
```
- 异或的应用：以异同为分类依据
  - 真值表

| XOR | 0 | 1 |
|-----|---|---|
| 0   | 0 | 1 |
| 1   | 1 | 0 |
# Code
```cpp
#include <iostream>
#include <vector>

using namespace std;

struct Toy
{
    int ori;
    string name;
};

vector<Toy> toys;

//R —Classify—> [0, R]
int num_mapping_from_zero(int r, int n)
{
    int len = r + 1;
    if (n >= 0) return n % len; //正数直接 mod
    else
    {
        return (n % len + len) % len;
        /*
        //负数取模，向0取整所以其结果要补一个len，使余数r为正
        if (n % len == 0) return 0; //特判如果能整除，余数是 0 而不是 len
        return n % len + len; //补len
        */
    }
}

//R —Classify—> [l, R]
int num_mapping(int l, int r, int n)
{
    //左端点偏移
    int t = n - l;
    return num_mapping_from_zero(r - l, t) + l;
}

int main()
{
    int n, m;
    cin >> n >> m;
    while (n --)
    {
        int ori;
        string name;
        cin >> ori >> name;
        toys.push_back({ori, name});
    }

    int now = 0;
    while (m --)
    {
        int dir, step;
        cin >> dir >> step;
        //运用异或，相同往顺时针左走，相反逆时针往右走
        //相同的话，step取负，否则取正
        if (!(toys[now].ori ^ dir))
            step = -step;
        //mod运算：将实数域R映射到一个区间内
        now = num_mapping(0, toys.size() - 1, now + step);
    }

    cout << toys[now].name << endl;
    return 0;
}
```
