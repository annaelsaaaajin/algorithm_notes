- [P2670 [NOIP2015 普及组] 扫雷游戏](https://www.luogu.com.cn/problem/P2670)
# 解题思路
- 偏移量数组
    - `int dx[]`
    - `int dy[]` 
- 处理越界
    - 构造虚拟边界
        - 循环从 1 开始
        - 自己填边界符号 
    -  对偏移后的坐标进行范围判断
# Code
- 判断范围
```cpp
#include <iostream>

using namespace std;

const int N = 110;

//偏移量数组
int dx[] = {0, 1, 1, 1, 0, -1, -1, -1};
int dy[] = {1, 1, 0, -1, -1, -1, 0, 1};

char mat[N][N];
int res[N][N];
int m, n;

void get_mine(int x, int y)
{
    int a, b;
    // 如果是地雷则标记为地雷
    if (mat[x][y] == '*')
    {
        res[x][y] = -1;
        return;
    }
    
    int cnt = 0;
    //枚举8个方向
    for (int i = 0; i < 8; i ++)
    {
        a = x + dx[i], b = y + dy[i];
        //保证不越界，而且是雷，计数加1
        if (a >= 0 && a < n && b >= 0 && b <= m && mat[a][b] == '*')
            cnt ++;
    }
    res[x][y] = cnt;
}

int main()
{
    cin >> n >> m;
    for (int i = 0; i < n; i ++)
        for (int j = 0; j < m; j ++)
            cin >> mat[i][j];

    for (int i = 0; i < n; i ++)
        for (int j = 0; j < m; j ++)
            get_mine(i, j);

    for (int i = 0; i < n; i ++)
    {
        for (int j = 0; j < m; j ++)
        {
            if (res[i][j] == -1) cout << '*';
            else cout << res[i][j];
        }
        cout << endl;
    }

    return 0;
}
```

- 虚拟边界
  - 从1开始循环，0默认是边界 

```cpp
#include <iostream>

using namespace std;

const int N = 110;

//偏移量数组
int dx[] = {0, 1, 1, 1, 0, -1, -1, -1};
int dy[] = {1, 1, 0, -1, -1, -1, 0, 1};

char mat[N][N];
int res[N][N];
int m, n;

void get_mine(int x, int y)
{
    int a, b;
    // 如果是地雷则标记为地雷
    if (mat[x][y] == '*')
    {
        res[x][y] = -1;
        return;
    }
    
    int cnt = 0;
    //枚举8个方向
    for (int i = 0; i < 8; i ++)
    {
        a = x + dx[i], b = y + dy[i];
        //如果是雷，计数加1
        //如果 a == 0 || b == 0 全局变量初始化默认为0不是'*'
        //也满足判断条件
        if (mat[a][b] == '*')
            cnt ++;
    }
    res[x][y] = cnt;
}

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i ++)
        for (int j = 1; j <= m; j ++)
            cin >> mat[i][j];

    for (int i = 1; i <= n; i ++)
        for (int j = 1; j <= m; j ++)
            get_mine(i, j);

    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= m; j ++)
        {
            if (res[i][j] == -1) cout << '*';
            else cout << res[i][j];
        }
        cout << endl;
    }

    return 0;
}
```
