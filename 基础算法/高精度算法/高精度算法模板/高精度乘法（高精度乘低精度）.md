# 高精度乘法模板
$高精度 \times  低精度$
**主体**
```cpp
vector<int> mul(vector<int> &A, int b)
{
    //任何数乘0等于0
    if (b == 0 || (A.size() == 1 && A[0] == 0)) return {0};
    vector<int> C;
    //进位变量
    int t = 0;
    for (int i = 0; i < A.size(); i ++)
    {
        //每位相乘
        t += A[i] * b;
        //更新C[i]的值
        C.push_back(t % 10);
        //更新进位值
        t /= 10;
    }
    //处理剩余的进位
    while (t) C.push_back(t % 10), t /= 10;
    return C;
}
```

**模拟**
- 每位相乘
- 更新`C[i]`的值
- 更新进位值`t`

**细节**
- 任何数乘 $0$ 等于 $0$

  ```cpp
     return {0};
  ``` 

# Code
```cpp
#include <iostream>
#include <vector>

using namespace std;

vector<int> str_to_vec(string &s)
{
    vector<int> res;
    for (int i = s.size() - 1; i >= 0; i --) res.push_back(s[i] - '0');
    return res;
}

string vec_to_str(vector<int> &v)
{
    string res;
    for (int i = v.size() - 1; i >= 0; i --) res += (v[i] + '0');
    return res;
}

vector<int> mul(vector<int> &A, int b)
{
    if (b == 0 || (A.size() == 1 && A[0] == 0)) return {0};
    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size(); i ++)
    {
        t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    while (t) C.push_back(t % 10), t /= 10;
    return C;
}

int main()
{
    string a, c;
    int b;
    cin >> a >> b;
    vector<int> A, C;
    A = str_to_vec(a);
    C = mul(A, b);
    c = vec_to_str(C);
    cout << c;
    return 0;
}
```
